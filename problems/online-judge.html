#include <iostream>

using namespace std;
int prime(long long int n){
	if(n < 2){
		return 0;
	}
	if(n == 2){
		return 1;
	}
	if(n % 2 == 0){
		return 0;
	}
	for(long long int i = 3; i <= n / 2; i += 2){
		if(n % i == 0){
			return 0;
		}
	}
	return 1;
}
int main(void){
    long long int n;
    while(1){
    	cin >> n;
    	if(n == 0){
    		break;
    	}
    	if(1 == prime(n)){

    	}
    	else{
    		cout << "Compound Number";
    	}
    }
    return 0;
    }
