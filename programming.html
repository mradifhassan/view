#include <stdio.h>

int main(void){
  int m;
  scanf("%d", &m);
  int array[m];
  for(int i = 0; i < m; i++){
    scanf("%d", &array[i]);
  }
  int max = array[0];
  int size = sizeof(array) / sizeof(array[0]);
  for(int i = 1; i < size; i++){
    if(array[i] > max){
        max = array[i];
    }
    else{
      max = max;
    }
}
  printf("%d", max);
  return 0;
}

    int n, count = 0;
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) {
        if (n % i == 0) {
            count++;
        }
    }
    printf("%d", count - 1);
